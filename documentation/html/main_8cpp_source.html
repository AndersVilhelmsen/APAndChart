115// Initialize WiFi
116bool initWiFi() {
117 if(ssid=="" || ip==""){
118 Serial.println("Undefined SSID or IP address.");
119 return false;
120 }
121
122 WiFi.mode(WIFI_STA);
123 localIP.fromString(ip.c_str());
124 localGateway.fromString(gateway.c_str());
125
126
127 if (!WiFi.config(localIP, localGateway, subnet)){
128 Serial.println("STA Failed to configure");
129 return false;
130 }
131 WiFi.begin(ssid.c_str(), pass.c_str());
132 Serial.println("Connecting to WiFi...");
133
134 unsigned long currentMillis = millis();
135 previousMillis = currentMillis;
136
137 while(WiFi.status() != WL_CONNECTED) {
138 currentMillis = millis();
139 if (currentMillis - previousMillis >= interval) {
140 Serial.println("Failed to connect.");
141 return false;
142 }
143 }
144
145 Serial.println(WiFi.localIP());
146 return true;
147}
